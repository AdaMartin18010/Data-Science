class Iterator(ABC):
    @abstractmethod
    def has_next(self) -> bool:
        pass

    @abstractmethod
    def next(self) -> Any:
        pass

    @abstractmethod
    def get_iterator(self) -> Iterator:
        pass

    def __init__(self):
        self.names = ["Robert", "John", "Julie", "Lora"]

    def get_iterator(self) -> Iterator:

    def next(self) -> Any:

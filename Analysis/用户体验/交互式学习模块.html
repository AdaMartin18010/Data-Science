查询优化算法

-- 查询优化示例
EXPLAIN (ANALYZE, BUFFERS) 
SELECT u.name, p.title, c.content
FROM users u
JOIN posts p ON u.id = p.user_id
JOIN comments c ON p.id = c.post_id
WHERE u.active = true 
  AND p.created_at > '2024-01-01'
ORDER BY p.created_at DESC
LIMIT 10;

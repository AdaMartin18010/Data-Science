def quantize_to_int4(self, weights: np.ndarray) -> Tuple[np.ndarray, float, float]:
    """å°†FP32æƒé‡é‡åŒ–åˆ°INT4"""
    # è®¡ç®—é‡åŒ–å‚æ•°
    scale = np.max(np.abs(weights)) / 7.0  # INT4èŒƒå›´: -8åˆ°7
    zero_point = 0

    # é‡åŒ–
    quantized = np.round(weights / scale).astype(np.int8)

def pack_int4(self, values: np.ndarray) -> np.ndarray:
    """å°†INT4å€¼æ‰“åŒ…åˆ°INT8"""
    # å°†å€¼è½¬æ¢ä¸º0-15èŒƒå›´

    # æ‰“åŒ…ï¼šæ¯ä¸¤ä¸ªå€¼æ‰“åŒ…æˆä¸€ä¸ªå­—èŠ‚
    packed = np.zeros((len(values) + 1) // 2, dtype=np.uint8)
    for i in range(0, len(values), 2):
        if i + 1 < len(values):
            packed[i // 2] = (values[i] << 4) | values[i + 1]
        else:
            packed[i // 2] = values[i] << 4
    return packed

            INSERT OR REPLACE INTO ai_models

            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)

            model_id, model_name, version, model_data,
            str(metadata) if metadata else None,
            file_size, checksum, int(time.time()), int(time.time())

        """ä»SQLiteåŠ è½½æ¨¡å‹"""
